select I.SUMMARY, I.DESCRIPTION, IL.NAME
from ISSUES I
join ISSUESLISTS IL on IL.ID = I.ISSUESLISTS_ID
;

select I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
from ISSUES I
join USERS U on U.ID = I.USER_ID_ASSIGNEDTO
order by I.ID
;

select U.FIRSTNAME, U.LASTNAME, count(*) as ISSUESS_NUMBER
from USERS U join ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
group by I.USER_ID_ASSIGNEDTO
order by U.ID
;
